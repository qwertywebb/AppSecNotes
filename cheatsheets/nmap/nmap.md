# Сервисы и версии с установкой TCP-соединения:
-sV

# Запуск скриптов:
-sC

# -Pn не проверять жив ли хост(через ICMP ping), сразу сканировать порты

# ARP сканирование, без сканирования портов:
nmap -PR -sn TARGETS

# Ping сканирование, без сканирования портов:
nmap -PE -sn TARGETS

# ICMP timestamp сканирование, без сканирования портов(Если firewall блокирует echo запросы):
sudo nmap -PP -sn TARGETS

# ICMp netmask сканирование, без сканирования портов(Если firewall блокирует echo запросы):
sudo nmap -PM -sn TARGETS

# Установка TCP-соединения syn,syn-ack,ack:
-sT

# Без установки TCP-соединения, только syn:
-sS

# UDP-сканирование без установки TCP-соединения: отправка UDP-пакета на открытый порт ничего нам не даст, на закрытый вернется ошибка
-sU

# Скорость сканированиея:
-T5 - быстрое
-T1 - стандарт
-T0 - 1 порт за раз и ждёт 5 мин.

# Фрагментация пакета
-f

# Кол-во пакетов в секунду:
--max-rate 10
--min-rate 100

# Параллелизм:
--min-parallelism 500 - попытка "продавить" сеть скоростью (рискуя быть обнаруженным)

--max-parallelism 1 — идеальная скрытность для очень важных целей.

nmap -sC -sV 10.82.173.73 -p 22
