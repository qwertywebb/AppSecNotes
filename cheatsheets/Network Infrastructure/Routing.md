# Показать все маршруты
ip route show

# Показать какой интерфейс для конкретного IP
ip route get <IP>

# Добавить маршрут
sudo ip route add 192.168.100.0/24 via 192.168.21.1 dev tun0

# Удалить маршрут
sudo ip route del 192.168.12.0/24 dev tun0

# metric — приоритет маршрута (чем меньше, тем выше приоритет)
